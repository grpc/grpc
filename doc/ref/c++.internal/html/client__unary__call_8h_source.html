48 // Wrapper that performs a blocking unary call
49 template <class InputMessage, class OutputMessage>
50 Status BlockingUnaryCall(Channel* channel, const RpcMethod& method,
51  ClientContext* context, const InputMessage& request,
52  OutputMessage* result) {
53  CompletionQueue cq;
54  Call call(channel->CreateCall(method, context, &cq));
55  CallOpSet<CallOpSendInitialMetadata, CallOpSendMessage,
56  CallOpRecvInitialMetadata, CallOpRecvMessage<OutputMessage>,
57  CallOpClientSendClose, CallOpClientRecvStatus> ops;
58  Status status = ops.SendMessage(request);
59  if (!status.ok()) {
60  return status;
61  }
62  ops.SendInitialMetadata(context->send_initial_metadata_);
63  ops.RecvInitialMetadata(context);
64  ops.RecvMessage(result);
65  ops.ClientSendClose();
66  ops.ClientRecvStatus(context, &status);
67  call.PerformOps(&ops);
68  GPR_ASSERT((cq.Pluck(&ops) && ops.got_message) || !status.ok());
69  return status;
70 }
