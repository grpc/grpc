97 static __inline gpr_atm gpr_atm_no_barrier_fetch_add(gpr_atm *p,
98  gpr_atm delta) {
99  /* Use the CAS operation to get pointer-sized fetch and add */
100  gpr_atm old;
101  do {
102  old = *p;
103  } while (!gpr_atm_no_barrier_cas(p, old, old + delta));
104  return old;
105 }
