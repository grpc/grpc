60  ProtoBufferWriter(ByteBuffer* byte_buffer, int block_size, int total_size)
61  : block_size_(block_size),
62  total_size_(total_size),
63  byte_count_(0),
64  have_backup_(false) {
65  GPR_ASSERT(!byte_buffer->Valid());
67  grpc_byte_buffer* bp = grpc_raw_byte_buffer_create(nullptr, 0);

72  ~ProtoBufferWriter() override {
73  if (have_backup_) {
74  grpc_slice_unref(backup_slice_);
75  }
76  }

80  bool Next(void** data, int* size) override {
81  // Protobuf should not ask for more memory than total_size_.
82  GPR_ASSERT(byte_count_ < total_size_);
83  // 1. Use the remaining backup slice if we have one
84  // 2. Otherwise allocate a slice, up to the remaining length needed
85  // or our maximum allocation size
86  // 3. Provide the slice start and size available
87  // 4. Add the slice being returned to the slice buffer
88  size_t remain = static_cast<size_t>(total_size_ - byte_count_);
89  if (have_backup_) {
91  slice_ = backup_slice_;
92  have_backup_ = false;
93  if (GRPC_SLICE_LENGTH(slice_) > remain) {
94  GRPC_SLICE_SET_LENGTH(slice_, remain);
95  }
96  } else {
97  // When less than a whole block is needed, only allocate that much.
98  // But make sure the allocated slice is not inlined.
99  size_t allocate_length =
100  remain > static_cast<size_t>(block_size_) ? block_size_ : remain;
101  slice_ = grpc_slice_malloc(allocate_length > GRPC_SLICE_INLINED_SIZE
102  ? allocate_length
103  : GRPC_SLICE_INLINED_SIZE + 1);
104  }
105  *data = GRPC_SLICE_START_PTR(slice_);
106  // On win x64, int is only 32bit
107  GPR_ASSERT(GRPC_SLICE_LENGTH(slice_) <= INT_MAX);
108  byte_count_ += * size = static_cast<int>(GRPC_SLICE_LENGTH(slice_));
109  // Using grpc_slice_buffer_add could modify slice_ and merge it with the
110  // previous slice. Therefore, use grpc_slice_buffer_add_indexed method to
111  // ensure the slice gets added at a separate index. It can then be kept
112  // around and popped later in the BackUp function.
113  grpc_slice_buffer_add_indexed(slice_buffer_, slice_);
114  return true;
115  }
116 
120  void BackUp(int count) override {
121  // count == 0 is invoked by ZeroCopyOutputStream users indicating that any
122  // potential buffer obtained through a previous call to Next() is final.
123  // ZeroCopyOutputStream implementations such as streaming output can use
124  // these calls to flush any temporary buffer and flush the output. The logic
125  // below is not robust against count == 0 invocations, so directly return.
126  if (count == 0) return;
127 
132  GPR_ASSERT(count <= static_cast<int>(GRPC_SLICE_LENGTH(slice_)));
133  grpc_slice_buffer_pop(slice_buffer_);
134  if (static_cast<size_t>(count) == GRPC_SLICE_LENGTH(slice_)) {
135  backup_slice_ = slice_;
136  } else {
137  backup_slice_ =
138  grpc_slice_split_tail(&slice_, GRPC_SLICE_LENGTH(slice_) - count);
139  grpc_slice_buffer_add(slice_buffer_, slice_);
140  }
141  // It's dangerous to keep an inlined grpc_slice as the backup slice, since
142  // on a following Next() call, a reference will be returned to this slice
143  // via GRPC_SLICE_START_PTR, which will not be an address held by
144  // slice_buffer_.
145  have_backup_ = backup_slice_.refcount != nullptr;
146  byte_count_ -= count;
147  }
148 
150  int64_t ByteCount() const override { return byte_count_; }
