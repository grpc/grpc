46 /* The clocks we support. */
47 typedef enum {
48  /* Monotonic clock. Epoch undefined. Always moves forwards. */
49  GPR_CLOCK_MONOTONIC = 0,
50  /* Realtime clock. May jump forwards or backwards. Settable by
51  the system administrator. Has its epoch at 0:00:00 UTC 1 Jan 1970. */
52  GPR_CLOCK_REALTIME,
53  /* CPU cycle time obtained by rdtsc instruction on x86 platforms. Epoch
54  undefined. Degrades to GPR_CLOCK_REALTIME on other platforms. */
55  GPR_CLOCK_PRECISE,
56  /* Unmeasurable clock type: no base, created by taking the difference
57  between two times */
58  GPR_TIMESPAN
59 } gpr_clock_type;
