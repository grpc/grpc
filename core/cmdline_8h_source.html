43 /* Simple command line parser.
44 
45  Supports flags that can be specified as -foo, --foo, --no-foo, -no-foo, etc
46  And integers, strings that can be specified as -foo=4, -foo blah, etc
47 
48  No support for short command line options (but we may get that in the
49  future.)
50 
51  Usage (for a program with a single flag argument 'foo'):
52 
53  int main(int argc, char **argv) {
54  gpr_cmdline *cl;
55  int verbose = 0;
56 
57  cl = gpr_cmdline_create("My cool tool");
58  gpr_cmdline_add_int(cl, "verbose", "Produce verbose output?", &verbose);
59  gpr_cmdline_parse(cl, argc, argv);
60  gpr_cmdline_destroy(cl);
61 
62  if (verbose) {
63  gpr_log(GPR_INFO, "Goodbye cruel world!");
64  }
65 
66  return 0;
67  } */
68 
69 typedef struct gpr_cmdline gpr_cmdline;
